purge_normal = {
	pop_escape_chance = 0.05
	pop_decline_rate = 100 # 1 pops per month, overrides BASE_POP_DECLINE, will only work on purge types																								 
	
	pop_group_modifier = {
		pop_happiness = -10
	}	
	
	potential = {
		exists = from
		from = { NOT = { has_valid_civic = civic_hive_devouring_swarm } }
	}
	
	allow = {
		hidden_trigger = { exists = from }
		if = { 
			limit = { is_lonely_hive_mind_pop = yes } 
			always = yes
		}
		else = {
			custom_tooltip = {
				fail_text = PURGES_ALLOWED
				OR = {
                    AND = {
                        # is_synthetic_species = yes #TODO Replace in 3.0.2
                        OR = {
                            has_trait = trait_photonic
                            has_trait = trait_mechanical
                            has_trait = trait_machine_unit
                            has_trait = trait_chromodynamic_power_module
                        }
                        from = { NOT = { has_policy_flag = ai_full_rights } }
                    }
                    from = {
                        OR = {
                            has_ethic = ethic_gestalt_consciousness
                            has_valid_civic = civic_fanatic_purifiers
                            has_policy_flag = purge_allowed
                        }
                    }
                }
			}	
		}
	}
	ai_will_do = {
		factor = 3
		modifier = {
			factor = 0
			from = {
				has_valid_ai_personality = yes
				NOT = {
					has_ai_personality_behaviour = purger
				}
			}
		}		
	}
}

purge_displacement = {
	pop_escape_chance = 1.0
	pop_decline_rate = 25 # 1 pop per 4 months, overrides BASE_POP_DECLINE, will only work on purge types																								  
	
	pop_group_modifier = {
		pop_happiness = -0.3
		pop_housing_usage_mult = -0.50
	}
    
    pop_tooltip = DISPLACING
	
	potential = {
		exists = from
		from = {
			NOR = {
				has_valid_civic = civic_hive_devouring_swarm
				has_valid_civic = civic_machine_terminator
				has_valid_civic = civic_fanatic_purifiers
				has_valid_civic = civic_assimilators ### Borg won't displace population
				has_valid_civic = civic_the_weak_will_perish
			}
		}
	}
	
	allow = {
		hidden_trigger = { exists = from }
		custom_tooltip = {
			fail_text = "lonely_hive_mind_pops_kill_themselves_purge_type"
			is_lonely_hive_mind_pop = no
		}
		custom_tooltip = {
			fail_text = "disconnected_machine_pops_kill_themselves_citizenship"
			is_disconnected_machine_pop = no
		}
	}
	ai_will_do = {
		factor = 1
		modifier = {
			factor = 10
			from = {
				has_authority = auth_hive_mind			
				NOT = { has_valid_civic = civic_hive_strength_of_legions }
			}
		}
		modifier = {
			factor = 10
			from = {
				has_authority = auth_machine_intelligence
			}
		}			
	}
}
