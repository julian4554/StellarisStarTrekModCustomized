		category = ships
		upkeep = { crew = @default_weapon_platform_crew_upkeep }

		sh_weapon_platform_base_stats = 1
	}
	combat_disengage_chance = @weapon_platform_disengagement_chance

		ship_size_value = @weapon_platform_size

        ship_class_value = @shipclass_military_special_value
		ship_tier_value = 5
	}
	is_space_object = yes
	size_multiplier = 4
	fleet_slot_size = 3
	num_target_locators = 2
	is_space_station = no

	base_buildtime = @cruiser_base_buildtime_5
	can_have_federation_design = no
	enable_default_design = yes
	default_behavior = line
	prerequisites = { "tech_xindi_probe_1" }
	potential_country = { can_design_ship = { ship = "xindi_probe" } }
	class = shipclass_military_special
	ship_roles = { brawler gunship explosive carrier screen artillery }
	construction_type = starbase_shipyard
	section_slots = { "mid" = { locator = "part1" } }
	
	required_component_set = "required_sensor_cruiser"
	required_component_set = "required_warp_cruiser"
	required_component_set = "required_impulse_cruiser"
	required_component_set = "required_hull_cruiser"	
	required_component_set = "required_shield_cruiser"
	required_component_set = "required_unique_cruiser" 
	required_component_set = "required_countermeasures_cruiser"
    required_component_set = "required_mission_cruiser"

		can_construct_ship = { ship = "xindi_probe" }
        owner = { planet_killers_banned = no }
	}
	ai_ship_data = {
		min = 1
        max = 2
    }
}
